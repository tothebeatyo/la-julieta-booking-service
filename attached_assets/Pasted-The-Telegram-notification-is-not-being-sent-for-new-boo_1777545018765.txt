STEP 3 — Update telegramNotifier.ts to include
phone number and note in the message:

export async function notifyBooking(details: {
  name: string;
  service: string;
  date: string;
  time: string;
  status: "success" | "failed";
  psid: string;
  note?: string;
}): Promise<void> {
  if (!TELEGRAM_BOT_TOKEN || !TELEGRAM_CHAT_ID) {
    console.warn("Telegram not configured");
    return;
  }

  const emoji = details.status === "success" ? "✅" : "⚠️";
  const noteText = details.note ? `\n📝 ${details.note}` : "";

  const message = `
${emoji} *La Julieta Booking Alert*

👤 *Name:* ${details.name}
💆 *Service:* ${details.service}
📅 *Date:* ${details.date}
🕐 *Time:* ${details.time}
🆔 *PSID:* ${details.psid}${noteText}
🔗 https://www.lajulieta.anypluspro.com
  `.trim();

  try {
    const response = await fetch(
      `https://api.telegram.org/bot${TELEGRAM_BOT_TOKEN}/sendMessage`,
      {
        method: "POST",
        headers: { "Content-Type": "application/json" },
        body: JSON.stringify({
          chat_id: TELEGRAM_CHAT_ID,
          text: message,
          parse_mode: "Markdown",
        }),
      }
    );

    if (!response.ok) {
      const err = await response.json();
      console.error("Telegram API error:", err);
    } else {
      console.log("Telegram notification sent successfully");
    }
  } catch (err) {
    console.error("Telegram notification failed:", err);
  }
}
