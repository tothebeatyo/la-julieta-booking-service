async function gotoWithRetry(
  page: Page, 
  url: string, 
  maxRetries = 3
): Promise<void> {
  for (let i = 0; i < maxRetries; i++) {
    try {
      await page.goto(url, { 
        waitUntil: "domcontentloaded",
        timeout: 60000
      });
      await page.waitForTimeout(2000);
      logger.info({ url, attempt: i + 1 }, "Page loaded successfully");
      return;
    } catch (err) {
      logger.warn({ url, attempt: i + 1, err }, "Navigation attempt failed, retrying...");
      if (i === maxRetries - 1) throw err;
      await page.waitForTimeout(3000);
    }
  }
}
