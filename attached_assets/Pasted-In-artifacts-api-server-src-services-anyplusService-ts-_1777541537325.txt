async function registerNewPatient(
  page: Page, 
  details: BookingDetails
): Promise<void> {
  const nameParts = details.name.trim().split(" ");
  const firstName = nameParts[0] ?? details.name;
  const lastName = nameParts.slice(1).join(" ") || "-";

  // Wait for register form
  await page.waitForSelector(
    'input[placeholder="First name"]', 
    { visible: true, timeout: 10000 }
  );

  // Fill first and last name
  await page.type('input[placeholder="First name"]', firstName);
  await page.type('input[placeholder="Last name"]', lastName);

  // Fill phone if available
  if (details.phone) {
    await page.evaluate((phone) => {
      const inputs = Array.from(document.querySelectorAll("input"));
      const mobileInput = inputs.find(i => 
        i.placeholder?.includes("9XX") || 
        i.type === "tel" ||
        i.placeholder?.toLowerCase().includes("mobile") ||
        i.placeholder?.toLowerCase().includes("phone")
      );
      if (mobileInput) {
        mobileInput.value = phone;
        mobileInput.dispatchEvent(new Event("input", { bubbles: true }));
        mobileInput.dispatchEvent(new Event("change", { bubbles: true }));
      }
    }, details.phone);
  }

  // Select Lead Source — Facebook
  await page.evaluate(() => {
    const selects = Array.from(document.querySelectorAll("select"));
    for (const select of selects) {
      const container = select.closest("div")?.textContent?.toLowerCase() ?? "";
      if (container.includes("lead source") || container.includes("how did")) {
        const fbOption = Array.from(select.options).find(o => 
          /facebook|social|online|instagram/i.test(o.text)
        );
        if (fbOption) select.value = fbOption.value;
        else if (select.options.length > 1) select.selectedIndex = 1;
        select.dispatchEvent(new Event("change", { bubbles: true }));
        break;
      }
    }
  });

  // Add notes
  await page.evaluate((details) => {
    const textareas = Array.from(document.querySelectorAll("textarea"));
    const notes = textareas.find(t => 
      t.placeholder?.toLowerCase().includes("note") ||
      t.placeholder?.toLowerCase().includes("additional")
    );
    if (notes) {
      notes.value = `Booked via Facebook Messenger. Service: ${details.service}. Date: ${details.date} at ${details.time}. Client type: ${details.clientType ?? "New"}.`;
      notes.dispatchEvent(new Event("input", { bubbles: true }));
    }
  }, details);

  await takeScreenshot(page, "register-form-filled");

  // Click Register Patient button
  await page.evaluate(() => {
    const buttons = Array.from(document.querySelectorAll("button"));
    const btn = buttons.find(b => 
      /register patient|register/i.test(b.textContent ?? "")
    );
    if (btn) btn.click();
  });

  await page.waitForTimeout(3000);
  await takeScreenshot(page, "patient-registered");
  logger.info("Patient registration submitted");
}
