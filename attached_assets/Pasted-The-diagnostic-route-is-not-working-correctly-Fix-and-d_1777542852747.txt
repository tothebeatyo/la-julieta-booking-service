// Add after server starts listening:
async function runStartupDiagnostic(): Promise<void> {
  try {
    const { pool } = await import("@workspace/db");
    
    // Check tables exist
    const tables = await pool.query(`
      SELECT table_name 
      FROM information_schema.tables 
      WHERE table_schema = 'public'
      ORDER BY table_name
    `);
    logger.info({ tables: tables.rows.map(r => r.table_name) }, "DB TABLES");

    // Check messages table
    const msgCount = await pool.query("SELECT COUNT(*) as total FROM messages");
    logger.info({ count: msgCount.rows[0] }, "MESSAGES COUNT");

    // Check clients table  
    const clientCount = await pool.query("SELECT COUNT(*) as total FROM clients");
    logger.info({ count: clientCount.rows[0] }, "CLIENTS COUNT");

    // Check recent messages
    const recentMsgs = await pool.query(
      "SELECT * FROM messages ORDER BY created_at DESC LIMIT 5"
    );
    logger.info({ messages: recentMsgs.rows }, "RECENT MESSAGES");

    // Create messages table if missing
    await pool.query(`
      CREATE TABLE IF NOT EXISTS messages (
        id SERIAL PRIMARY KEY,
        psid TEXT NOT NULL,
        direction TEXT NOT NULL,
        content TEXT NOT NULL,
        created_at TIMESTAMPTZ DEFAULT NOW()
      );
      CREATE INDEX IF NOT EXISTS idx_messages_psid ON messages(psid);
    `);
    logger.info("Messages table ensured");

  } catch (err) {
    logger.error({ err }, "STARTUP DIAGNOSTIC FAILED");
  }
}

import { pool } from "@workspace/db";

async function fixDb() {
  // Create messages table
  await pool.query(`
    CREATE TABLE IF NOT EXISTS messages (
      id SERIAL PRIMARY KEY,
      psid TEXT NOT NULL,
      direction TEXT NOT NULL 
        CHECK (direction IN ('inbound', 'outbound')),
      content TEXT NOT NULL,
      created_at TIMESTAMPTZ DEFAULT NOW()
    );
    CREATE INDEX IF NOT EXISTS idx_messages_psid 
      ON messages(psid);
    CREATE INDEX IF NOT EXISTS idx_messages_created 
      ON messages(created_at DESC);
  `);
  console.log("✅ Messages table created/verified");

  // Check clients table has all needed columns
  await pool.query(`
    ALTER TABLE clients 
    ADD COLUMN IF NOT EXISTS channel TEXT DEFAULT 'messenger';
    ALTER TABLE clients 
    ADD COLUMN IF NOT EXISTS last_message TEXT;
    ALTER TABLE clients 
    ADD COLUMN IF NOT EXISTS phone TEXT;
    ALTER TABLE clients 
    ADD COLUMN IF NOT EXISTS service TEXT;
    ALTER TABLE clients 
    ADD COLUMN IF NOT EXISTS name TEXT;
  `).catch(e => console.log("Column check:", e.message));
  
  console.log("✅ Clients table verified");
  
  // Count existing data
  const msgs = await pool.query("SELECT COUNT(*) FROM messages");
  const clients = await pool.query("SELECT COUNT(*) FROM clients");
  console.log("Messages:", msgs.rows[0].count);
  console.log("Clients:", clients.rows[0].count);
  
  await pool.end();
}

// CORRECT way to fetch messages:
const fetchMessages = async (psid: string) => {
  const token = localStorage.getItem("adminToken") 
    ?? sessionStorage.getItem("adminToken");
    
  const response = await fetch(
    `/api/admin/clients/${psid}/messages`,
    {
      headers: {
        "Authorization": `Bearer ${token}`,
        "Content-Type": "application/json",
      },
    }
  );
  
  if (!response.ok) {
    console.error("Failed to fetch messages:", response.status);
    return [];
  }
  
  const data = await response.json();
  console.log("Messages response:", data); // debug log
  return data.messages ?? [];
};
