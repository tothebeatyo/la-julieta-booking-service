const fetchMessages = async (clientPsid: string): Promise<void> => {
  try {
    setLoadingMessages(true);
    setMessagesError(null);
    
    const token = getToken();
    console.log("Fetching messages for:", clientPsid);
    console.log("Using token:", token ? "present" : "MISSING");
    
    const response = await fetch(
      `/api/admin/clients/${clientPsid}/messages`,
      {
        headers: {
          "Authorization": `Bearer ${token}`,
          "Content-Type": "application/json",
        },
      }
    );
    
    console.log("Response status:", response.status);
    
    if (!response.ok) {
      const errText = await response.text();
      console.error("API error:", errText);
      setMessagesError(`Failed to load: ${response.status}`);
      return;
    }
    
    const data = await response.json();
    console.log("Messages data:", data);
    console.log("Message count:", data.messages?.length);
    
    setMessages(data.messages ?? []);
    setSelectedClient(data.client ?? null);
    
  } catch (err) {
    console.error("Fetch error:", err);
    setMessagesError("Network error loading messages");
  } finally {
    setLoadingMessages(false);
  }
};

STEP 7 — Add manual reply box at bottom of chat
Below the messages list, add a reply input:

const [replyText, setReplyText] = useState("");
const [sendingReply, setSendingReply] = useState(false);

const sendReply = async () => {
  if (!replyText.trim() || !selectedPsid) return;
  setSendingReply(true);
  try {
    const token = getToken();
    const response = await fetch(
      `/api/admin/clients/${selectedPsid}/send-message`,
      {
        method: "POST",
        headers: {
          "Authorization": `Bearer ${token}`,
          "Content-Type": "application/json",
        },
        body: JSON.stringify({ message: replyText }),
      }
    );
    if (response.ok) {
      setReplyText("");
      await fetchMessages(selectedPsid); // refresh messages
    }
  } catch (err) {
    console.error("Send error:", err);
  } finally {
    setSendingReply(false);
  }
};

// Reply box JSX at bottom of chat panel:
<div className="border-t p-3 flex gap-2 bg-white">
  <input
    type="text"
    value={replyText}
    onChange={(e) => setReplyText(e.target.value)}
    onKeyDown={(e) => e.key === "Enter" && sendReply()}
    placeholder="Type a reply..."
    className="flex-1 border rounded-full px-4 py-2 text-sm focus:outline-none focus:border-pink-400"
    disabled={sendingReply}
  />
  <button
    onClick={sendReply}
    disabled={sendingReply || !replyText.trim()}
    className="bg-pink-500 text-white px-4 py-2 rounded-full text-sm disabled:opacity-50"
  >
    {sendingReply ? "..." : "Send"}
  </button>
</div>
