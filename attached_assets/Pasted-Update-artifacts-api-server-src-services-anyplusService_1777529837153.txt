    await page.click('button:has-text("Book"), button:has-text("Confirm"), button:has-text("Save"), button:has-text("Create Appointment")');

    const successIndicator = await page.locator(

    const finalScreenshot = await takeScreenshot(page, "14-final-result");

    // Update DB
    await pool.query(

       WHERE psid = $2`,

      error: err instanceof Error ? err.message : String(err)
    };
  } finally {

  }

// Helper: convert "10:00 AM" → "10:00", "2:30 PM" → "14:30"
function convertTo24Hr(timeStr: string): string {
  const [time, modifier] = timeStr.split(" ");
  let [hours, minutes] = time.split(":").map(Number);
  if (modifier === "PM" && hours !== 12) hours += 12;
  if (modifier === "AM" && hours === 12) hours = 0;
  return `${String(hours).padStart(2, "0")}:${String(minutes).padStart(2, "0")}`;
}
