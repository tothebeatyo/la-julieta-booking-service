<button
  onClick={async () => {
    try {
      const response = await fetch(
        "/api/admin/setup-persistent-menu",
        {
          method: "POST",
          headers: {
            Authorization: `Bearer ${getToken()}`,
          },
        }
      );
      const data = await response.json();
      if (response.ok) {
        alert("✅ Menu setup successful!");
      } else {
        alert(`❌ Menu error: ${JSON.stringify(data)}`);

      alert(`❌ Error: ${String(err)}`);
    }
  }}

  setLoadingMessages(true);

    const token = getToken();

    console.log("Token exists:", !!token);
    
    const res = await fetch(
      `/api/admin/clients/${psid}/messages`,
      { headers: { Authorization: `Bearer ${token}` } }
    );
    
    console.log("Response status:", res.status);
    const data = await res.json();
    console.log("Messages received:", data.messages?.length);
    console.log("Full data:", data);
    
    setMessages(data.messages ?? []);
    setSelectedClient(data.client ?? null);
  } catch (err) {
    console.error("fetchMessages error:", err);
  } finally {
    setLoadingMessages(false);
  }
};
