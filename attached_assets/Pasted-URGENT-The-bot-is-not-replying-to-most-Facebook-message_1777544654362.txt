STEP 3 — Add crash protection to processEvent
Wrap the ENTIRE processEvent function body in 
try-catch and send a fallback reply on ANY error:

async function processEvent(
  event: Record<string, unknown>,
  channel: "messenger" | "instagram" = "messenger"
): Promise<void> {
  const sender = (event["sender"] ?? event["from"]) as { id: string } | undefined;
  const psid = sender?.id;

  try {
    if (!psid) return;
    if (event["delivery"] || event["read"]) return;

    const message = event["message"] as Record<string, unknown> | undefined;
    const postback = event["postback"] as { payload?: string } | undefined;
    if (message?.["is_echo"]) return;

    const text = ((message?.["text"] as string) ?? "").trim();
    const quickReplyPayload = (message?.["quick_reply"] as { payload?: string })?.payload;
    const postbackPayload = postback?.payload;
    const payload = quickReplyPayload ?? postbackPayload;

    logger.info({ psid, channel, text, payload }, "📨 Processing event");

    // Set channel in session
    setSession(psid, { channel });

    // Log inbound message
    const inboundContent = text || payload || "(attachment)";
    logMessage(psid, "inbound", inboundContent).catch(() => {});
    upsertClient({ psid, lastMessage: inboundContent, status: "inquiry", channel }).catch(() => {});
    getProfileName(psid).then(name => {
      if (name) upsertClient({ psid, name, channel }).catch(() => {});
    }).catch(() => {});

    // No text and no payload = show welcome
    if (!text && !payload) {
      setSession(psid, { step: "choosing_intent" });
      await sendWithDelay(psid, randomPick(WELCOME_MESSAGES), 500);
      await sendWithDelayAndQuickReplies(psid, INTENT_MENU_TEXT, INTENT_QUICK_REPLIES, 800);
      return;
    }

    // Route to booking flow
    await handleBookingFlow(psid, text, payload);

  } catch (err) {
    logger.error({ err, psid }, "❌ processEvent CRASHED");

    // Always send fallback so client gets a reply
    if (psid) {
      try {
        await sendWithDelay(
          psid,
          "Hi! 💕 Thank you for messaging La Julieta Beauty Center! How can we help you today?",
          500
        );
        await sendWithDelayAndQuickReplies(
          psid,
          INTENT_MENU_TEXT,
          INTENT_QUICK_REPLIES,
          800
        );
      } catch (fallbackErr) {
        logger.error({ fallbackErr, psid }, "❌ Fallback message also failed");
      }
    }
  }
}
