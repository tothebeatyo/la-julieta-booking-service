import { chromium, type Browser, type Page } from "playwright";
import { logger } from "../lib/logger";
import { pool } from "@workspace/db";

const ANYPLUSPRO_URL = "https://www.lajulieta.anypluspro.com";
const USERNAME = process.env["ANYPLUSPRO_USERNAME"] ?? "";
const PASSWORD = process.env["ANYPLUSPRO_PASSWORD"] ?? "";
const SCREENSHOT_DIR = "logs/screenshots";

export interface BookingDetails {
  psid: string;
  name: string;
  service: string;
  date: string;
  time: string;
}

export interface BookingResult {
  success: boolean;
  screenshotPath?: string;
  error?: string;
}

async function takeScreenshot(page: Page, name: string): Promise<string> {
  const filename = `${SCREENSHOT_DIR}/${name}-${Date.now()}.png`;
  await page.screenshot({ path: filename, fullPage: true });
  logger.info({ filename }, "Screenshot saved");
  return filename;
}

export async function autoBook(details: BookingDetails): Promise<BookingResult> {
  let browser: Browser | null = null;
  try {
    logger.info({ details }, "Starting AnyPlusPro auto-booking");

    browser = await chromium.launch({
      headless: true,
      args: ["--no-sandbox", "--disable-setuid-sandbox"]
    });

    const page = await browser.newPage();
    page.setDefaultTimeout(30000);

    // Step 1: Go to login page
    await page.goto(ANYPLUSPRO_URL, { waitUntil: "networkidle" });
    await takeScreenshot(page, "01-landing");

    // Step 2: Fill login form
    await page.fill('input[type="email"], input[name="username"], input[name="email"]', USERNAME);
    await page.fill('input[type="password"]', PASSWORD);
    await takeScreenshot(page, "02-filled-login");

    // Step 3: Submit login
    await page.click('button[type="submit"], button:has-text("Login"), button:has-text("Sign in")');
    await page.waitForLoadState("networkidle");
    await takeScreenshot(page, "03-after-login");

    // Step 4: Check login success
    const currentUrl = page.url();
    if (currentUrl.includes("login")) {
      await takeScreenshot(page, "04-login-failed");
      throw new Error("Login failed — still on login page after submit");
    }

    // Step 5: Screenshot dashboard so we can see layout
    await takeScreenshot(page, "05-dashboard");
    logger.info("Login successful — screenshot taken of dashboard");

    // Update DB to pending
    await pool.query(
      `UPDATE clients SET anypluspro_status = 'pending', updated_at = NOW() WHERE psid = $1`,
      [details.psid]
    );

    // Booking form steps will be added after we review screenshots
    return {
      success: false,
      screenshotPath: "logs/screenshots/05-dashboard",
      error: "Booking form steps pending — check screenshots"
    };

  } catch (err) {
    logger.error({ err }, "AnyPlusPro auto-booking failed");
    return {
      success: false,
      error: err instanceof Error ? err.message : String(err)
    };
  } finally {
    if (browser) await browser.close();
  }
}

export async function retryAutoBooking(psid: string): Promise<{ success: boolean; error?: string }> {
  const result = await pool.query(
    "SELECT name, service, preferred_date, preferred_time FROM clients WHERE psid = $1",
    [psid]
  );
  if (!result.rows[0]) return { success: false, error: "Client not found" };
  const row = result.rows[0];
  return autoBook({
    psid,
    name: row.name,
    service: row.service,
    date: row.preferred_date,
    time: row.preferred_time,
  });
}
